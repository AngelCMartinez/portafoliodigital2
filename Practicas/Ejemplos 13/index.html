Ejemplo 13: Crear una lista HTML a partir de un array de objetos

    <!DOCTYPE html>
    <html lang="es">
    <head>
      <meta charset="UTF-8">
      <meta name="viewport" content="width=device-width, initial-scale=1.0">
      <title>Ejemplo 13: Crear Lista HTML</title>
    </head>
    <body>

      <h2>Ejemplo 13: Crear una lista HTML a partir de un array de objetos</h2>
      <button onclick="mostrarListaProductos()">Mostrar Lista de Productos</button>
      <ul id="listaProductos"></ul>

      <script>
        const productos = [
          { nombre: 'Manzana', precio: 1.5 },
          { nombre: 'Banana', precio: 0.8 }
        ];

        function mostrarListaProductos() {
          const listaProductos = productos.map(producto => `<li>${producto.nombre} - $${producto.precio}</li>`).join('');
          document.getElementById('listaProductos').innerHTML = listaProductos;
        }
      </script>

    </body>
    </html>
