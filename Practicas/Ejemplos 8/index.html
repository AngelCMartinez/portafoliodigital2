  <script>
    const nombres = ['Ana', 'Juan', 'Pedro'];
    function mostrarNombres() {
      let nombresTexto = '';
      nombres.forEach(nombre => {
        nombresTexto += nombre + ' ';
      });
      document.getElementById('nombresLista').innerText = nombresTexto.trim();
    }
  </script>
