<body>

  <h2>Ejemplo 19: Verificación de campos obligatorios</h2>
  <form id="miFormulario">
    <input type="text" name="nombre" placeholder="Nombre" required><br>
    <input type="email" name="correo" placeholder="Correo" required><br>
    <button type="button" onclick="verificarCampos()">Verificar Campos</button>
  </form>

  <script>
    function verificarCampos() {
      const formulario = document.getElementById('miFormulario');
      const camposVacios = Array.from(formulario.querySelectorAll('input:required')).filter(input => !input.value).map(input => input.name);
      if (camposVacios.length > 0) {
        alert(`Campos vacíos: ${camposVacios.join(', ' )}`);
      } else {
        alert('Todos los campos están completos');
      }
    }
  </script>
