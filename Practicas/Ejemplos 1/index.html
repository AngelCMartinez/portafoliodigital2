Ejemplo 1: Crear un contador con closures

<html lang="es">
<head>

  <title>Ejemplo 1: Closures y Contador</title>
</head>
<body>
  <h2>Ejemplo 1: Crear un contador con closures</h2>
  <button onclick="incrementarContador()">Incrementar Contador</button>
  <p id="contadorValor"></p>
  <script>
    function crearContador() {
      let contador = 0;
      return function() {
        contador++;
        return contador;
      };
    }
    const miContador = crearContador();
    function incrementarContador() {
      document.getElementById('contadorValor').innerText = miContador();
    }
  </script>
</body>
