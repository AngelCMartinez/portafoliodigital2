Ejemplo 20: Procesar Datos de una API

    <!DOCTYPE html>
    <html lang="es">
    <head>
      <meta charset="UTF-8">
      <meta name="viewport" content="width=device-width, initial-scale=1.0">
      <title>Procesamiento de datos JSON</title>
    </head>
    <body>

    <script>

    async function obtenerDatos() {
        try {
          const respuesta = await fetch('https://rickandmortyapi.com/api/character');
          if (!respuesta.ok) {
            throw new Error('Error al obtener los datos');
          }

          const datos = await respuesta.json();
          return datos.results.slice(0, 5).map(personaje => ({
            id: personaje.id,
            nombre: personaje.name,
            imagen: personaje.image
          }));
        } catch (error) {
          console.error('Error en obtenerDatos:', error);
          return [];
        }
    }

    async function obtenerDatosYMostrar() {
        const datos = await obtenerDatos();
        const listaDatos = document.getElementById('listaDatosProcesados');
        listaDatos.innerHTML = '';

        if (datos.length > 0) {
          const elementos = datos.map(dato =>
            `<li><strong>${dato.id}:</strong> ${dato.nombre} <br> <img src="${dato.imagen}" alt="${dato.nombre}" width="100"></li>`
          ).join('');

          listaDatos.innerHTML = elementos;
        } else {
          listaDatos.innerHTML = '<li>No se encontraron datos.</li>';
        }
    }

    </script>
      <h2>Ejemplo 20: Procesar Datos de una API</h2>
      <button onclick="obtenerDatosYMostrar()">Cargar Datos</button>
      <ul id="listaDatosProcesados"></ul>

    </body>
    </html>
