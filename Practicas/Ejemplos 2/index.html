  <h2>Ejemplo 2: Encapsulación con closures</h2>
  <button onclick="incrementarModulo()">Incrementar</button>
  <p id="moduloValor"></p>
  <script>
    function crearMódulo(valorInicial) {
      let valor = valorInicial;
      return {
        incrementar: function() {
          valor++;
        },
        obtenerValor: function() {
          return valor;
        }
      };
    }
    const miModulo = crearMódulo(10);
    function incrementarModulo() {
      miModulo.incrementar();
      document.getElementById('moduloValor').innerText = miModulo.obtenerValor();
    }
  </script>
</body>
