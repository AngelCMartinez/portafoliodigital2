Ejemplo 18: Filtrado de listas en tiempo real

    <!DOCTYPE html>
    <html lang="es">
    <head>
      <meta charset="UTF-8">
      <meta name="viewport" content="width=device-width, initial-scale=1.0">
      <title>Filtrado de listas en tiempo real</title>
    </head>
    <body>

      <h2>Ejemplo 18: Filtrado de listas en tiempo real</h2>
      <input type="text" id="busqueda" oninput="filtrarProductos()" placeholder="Buscar producto">
      <ul id="listaFiltrada"></ul>

      <script>
        const productos = [
          { nombre: 'Manzana' },
          { nombre: 'Banana' },
          { nombre: 'Pera' }
        ];

        function filtrarProductos() {
          const busqueda = document.getElementById('busqueda').value.toLowerCase();
          const lista = document.getElementById('listaFiltrada');
          lista.innerHTML = '';
          const productosFiltrados = productos.filter(producto => producto.nombre.toLowerCase().includes(busqueda));
          productosFiltrados.forEach(producto => {
            const li = document.createElement('li');
            li.textContent = producto.nombre;
            lista.appendChild(li);
          });
        }
      </script>

    </body>
    </html>
