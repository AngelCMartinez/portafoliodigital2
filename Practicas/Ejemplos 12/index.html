Ejemplo 12: Filtrado y ordenamiento de elementos

    <!DOCTYPE html>
    <html lang="es">
    <head>
      <meta charset="UTF-8">
      <meta name="viewport" content="width=device-width, initial-scale=1.0">
      <title>Filtrado y ordenamiento de elementos</title>
    </head>
    <body>

      <h2>Ejemplo 12: Filtrado y ordenamiento de elementos</h2>
      <button onclick="mostrarTareasPendientes()">Mostrar Tareas Pendientes</button>
      <ul id="listaTareas"></ul>

      <script>
        function mostrarTareasPendientes() {
          const tareas = [
            { id: 1, completada: false, descripcion: 'Tarea 1' },
            { id: 2, completada: true, descripcion: 'Tarea 2' }
        ];
        const lista = document.getElementById('listaTareas');
        lista.innerHTML = '';
        const tareasPendientes = tareas.filter(tarea => !tarea.completada)
          .map(tarea => `<li>${tarea.descripcion}</li>`)
          .join('');
        lista.innerHTML = tareasPendientes;
        }
      </script>

    </body>
    </html>
