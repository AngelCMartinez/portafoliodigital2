Ejemplo 6: Uso de filter en Arrays

Números pares:

<!DOCTYPE html>
<html lang="es">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Ejemplo filter en Arrays</title>
</head>
<body>

  <h2>Ejemplo: Uso de filter en Arrays</h2>
  <button onclick="mostrarPares()">Mostrar Números Pares</button>
  <p>Números pares: <span id="pares"></span></p>

  <script>
    const numeros = [1, 2, 3, 4, 5];
    function mostrarPares() {
      const pares = numeros.filter(numero => numero % 2 === 0);
      document.getElementById('pares').innerText = pares.join(', ');
    }
  </script>

</body>
</html>
