Ejemplo 10: Personas Mayores de 30

<!DOCTYPE html>
<html lang="es">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Personas Mayores de 30</title>
  <script>
    const personas = [
      { nombre: 'Ana', edad: 30 },
      { nombre: 'Juan', edad: 25 },
      { nombre: 'Pedro', edad: 35 }
    ];

    function filtrarMayoresDe30() {
      const nombresMayoresDe30 = personas
        .filter(persona => persona.edad > 30)
        .map(persona => persona.nombre);

      alert("Nombres de personas mayores de 30: " + nombresMayoresDe30.join(', '));
    }
  </script>
  <h1>Personas Mayores de 30</h1>
  <button onclick="filtrarMayoresDe30()">Mostrar Nombres Mayores de 30</button>
</head>
<body>
</body>
</html>
