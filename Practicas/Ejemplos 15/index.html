  <h2>Ejemplo 15: Crear una función que combine map y filter</h2>
  <button onclick="mostrarNumerosParesDoblados()">Mostrar Números Pares Doblados</button>
  <p>Números pares doblados: <span id="numerosParesDoblados"></span></p>

  <script>
    function mapearYFiltrar(array, mapeo, filtro) {
      return array.map(mapeo).filter(filtro);
    }

    const numeros = [1, 2, 3, 4, 5];
    const numerosParesDoblados = mapearYFiltrar(numeros, numero => numero * 2, numero => numero % 2 === 0);

    function mostrarNumerosParesDoblados() {
      document.getElementById('numerosParesDoblados').innerText = numerosParesDoblados.join(', ');
    }
  </script>
