Mapear con Prefijo

<!DOCTYPE html>
<html lang="es">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Mapear con Prefijo</title>
  <script>
    function mapearConPrefijo(array, prefijo) {
      return array.map(elemento => `${prefijo}-${elemento}`);
    }

    const numeros = [1, 2, 3];

    function mostrarMapeo() {
      const resultado = mapearConPrefijo(numeros, 'Número');
      alert("Resultado del mapeo: " + resultado.join(', '));
    }
  </script>
  <h1>Mapear con Prefijo</h1>
  <button onclick="mostrarMapeo()">Mapear Números con Prefijo</button>
</head>
<body>

</body>
</html>
