Ejemplo 9: Filtrar por Propiedad

<!DOCTYPE html>
<html lang="es">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Filtrar por Propiedad</title>
  <script>
    function filtrarPorPropiedad(array, propiedad, valor) {
      return array.filter(elemento => elemento[propiedad] === valor);
    }

    const personas = [
      { nombre: 'Ana', edad: 30 },
      { nombre: 'Juan', edad: 25 },
      { nombre: 'Pedro', edad: 35 }
    ];

    function mostrarFiltrado() {
      const resultado = filtrarPorPropiedad(personas, 'edad', 30);
      alert("Resultado del filtro: " + JSON.stringify(resultado));
    }
  </script>
  <h1>Filtrar por Propiedad</h1>
  <button onclick="mostrarFiltrado()">Filtrar por Edad 30</button>
</head>
<body>
</body>
</html>
