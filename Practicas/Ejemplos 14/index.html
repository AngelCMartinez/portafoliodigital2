    const numerosParesCuadrados = numeros.filter(num => num % 2 === 0).map(num => num * num);

    function mostrarNumerosParesCuadrados() {
      document.getElementById('numerosParesCuadrados').innerText = numerosParesCuadrados.join(', ');
    }
