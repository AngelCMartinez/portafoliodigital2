Ejemplo 3: Función de orden superior con closures

  <title>Ejemplo 3: Función de Orden Superior</title>
</head>
<body>
  <h2>Ejemplo 3: Función de orden superior con closures</h2>
  <button onclick="duplicarNumero()">Duplicar Número</button>
  <button onclick="triplicarNumero()">Triplicar Número</button>
  <p id="resultadoMultiplicador"></p>
  <script>
    function crearMultiplicador(factor) {
      return function(numero) {
        return numero * factor;
      };
    }
    const duplicar = crearMultiplicador(2);
    const triplicar = crearMultiplicador(3);
    function duplicarNumero() {
      document.getElementById('resultadoMultiplicador').innerText = duplicar(5);
    }
    function triplicarNumero() {
      document.getElementById('resultadoMultiplicador').innerText = triplicar(7);
    }
  </script>
</body>
