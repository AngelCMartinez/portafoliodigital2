Ejemplo 17:Agrupar y sumarizar datos

    <!DOCTYPE html>
    <html lang="es">
    <head>
      <meta charset="UTF-8">
      <meta name="viewport" content="width=device-width, initial-scale=1.0">
      <title>Agrupar y sumarizar datos</title>
    </head>
    <body>

      <h2>Agrupar y sumarizar datos</h2>
      <button onclick="mostrarVentas()">Mostrar Ventas</button>
      <ul id="listaVentas"></ul>

      <script>
        function mostrarVentas() {
          const ventas = [
            { producto: 'Manzana', cantidad: 10 },
            { producto: 'Banana', cantidad: 5 },
            { producto: 'Manzana', cantidad: 8 }
          ];
          const ventasPorProducto = ventas.reduce((acc, venta) => {
            acc[venta.producto] = (acc[venta.producto] || 0) + venta.cantidad;
            return acc;
          }, {});
          const lista = document.getElementById('listaVentas');
          lista.innerHTML = Object.entries(ventasPorProducto).map(([producto, cantidad]) => `

  • ${producto}: ${cantidad}
    • `).join('');
        }
      </script>

    </body>
    </html>
